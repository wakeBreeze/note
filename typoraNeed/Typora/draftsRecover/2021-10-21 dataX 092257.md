```json
{
    "job": {
        "setting": {
            "speed": {
                 "channel": 3
            },
            "errorLimit": {
                "record": 0,
                "percentage": 0.02
            }
        },
        "content": [
            {
                "reader": {
                    "name": "mysqlreader",
                    "parameter": {
                        "username": "root",
                        "password": "root",
                        "column": [
                            "deptno",
                            "dname",
                            "loc"
                        ],
                        "splitPk": "",
                        "connection": [
                            {
                                "table": [
                                    "dept"
                                ],
                                "jdbcUrl": [
     "jdbc:mysql://192.168.56.10:3306/test"
                                ]
                            }
                        ]
                    }
                },
               "writer": {
                    "name": "oraclewriter",
                    "parameter": {
                        "username": "system",
                        "password": "system",
                        "column": [
                            "deptno",
                            "dname",
                            "loc"
                        ],
                        "preSql": [
                            "delete from dept"
                        ],
                        "connection": [
                            {

                                "table": [
                                    "dept"
                                ]
                            }
                        ]
                    }
                }
            }
        ]
    }
}
```

***oracle2mysql.json***

```json
{
    "job": {
        "setting": {
            "speed": {
            	"channel": 5
            }
        },
        "content": [
            {
                "reader": {
                    "name": "oraclereader",
                    "parameter": {
                        "username": "system",
                        "password": "system",
                        "where": "",
                        "connection": [
                            {
                                "querySql": [
                                    "select deptno,dname,loc from dept where deptno < 30"
                                ],
                                "jdbcUrl": [
                                    "jdbc:oracle:thin:@192.168.56.10:1521:orcl"
                                ]
                            }
                        ]
                    }
                },
                "writer": {
                    "name": "mysqlwriter",
                    "parameter": {
                        "writeMode": "insert",
                        "username": "root",
                        "password": "root",
                        "column": [
                            "deptno",
                            "dname",
                            "loc"
                        ],
                        "session": [
                        	"set session sql_mode='ANSI'"
                        ],
                        "preSql": [
                            "delete from test"
                        ],
                        "connection": [
                            {
                                "jdbcUrl": "jdbc:mysql://192.168.56.10:3306/test?useUnicode=true&characterEncoding=gbk",
                                "table": [
                                    "d"
                                ]
                            }
                        ]
                    }
                }
            }
        ]
    }
}

```
